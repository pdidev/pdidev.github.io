379  {
380  // self-copy: nothing to do
381  if (&other == this) return *this;
382  // we'll be copied into, start nullifying ourselves first

544 
551  void* release() noexcept
552  {
553  if (is_null()) return nullptr;
554 
555  // notify everybody of the nullification
556  while (!m_content->m_buffer->m_notifications.empty()) {
557  // get the key of a notification
558  const Reference_base* key = m_content->m_buffer->m_notifications.begin()->first;
559  // call this notification, this might invalidate any iterator
560  m_content->m_buffer->m_notifications.begin()->second(*this);
561  // remove the notification we just called
562  m_content->m_buffer->m_notifications.erase(key);
563  }
564 
565  void* result = m_content->m_data;
566  m_content->m_data = nullptr;
567  m_content->m_buffer->m_delete = []() {}; // Referenced_metadata won't delete data
568 
569  unlink();
570 
571  return result;
572  }
573 
578  void on_nullify(std::function<void(Ref)> notifier) const noexcept
579  {
580  if (!is_null()) m_content->m_buffer->m_notifications[this] = notifier;
581  }

583 private:
591  bool PDI_NO_EXPORT is_null() const noexcept
592  {
593  if (!m_content) return true;
594  if (!m_content->m_data) {
595  unlink();
596  return true;
597  }
598  return false;
599  }
600 
605  void PDI_NO_EXPORT unlink() const noexcept
606  {
607  assert(m_content);
608  m_content->m_buffer->m_notifications.erase(this);
609  if (R || W) --m_content->m_buffer->m_write_locks;
610  if (W) --m_content->m_buffer->m_read_locks;
611  --m_content->m_owners;
612  if (!m_content->m_owners) delete m_content;
613  m_content = nullptr;
614  }
615 
623  void PDI_NO_EXPORT link(Referenced_data* content) noexcept
624  {
625  assert(!m_content);
626  if (!content || !content->m_data) return; // null ref
627  if (R && content->m_buffer->m_read_locks) return;
628  if (W && content->m_buffer->m_write_locks) return;
629  m_content = content;
630  ++m_content->m_owners;
631  if (R || W) ++m_content->m_buffer->m_write_locks;
632  if (W) ++m_content->m_buffer->m_read_locks;
633  }
