303  Ref_any(Ref_any&& other) noexcept:
304  Reference_base()
305  {
306  if (!other.m_content) return;
307  // the other ref notification disappears
308  other.m_content->m_buffer->m_notifications.erase(&other);

322  Ref_any(void* data, std::function<void(void*)> freefunc, Datatype_uptr type, bool readable, bool writable):
323  Reference_base()

326  throw Type_error{"Referencing null data with non-null size"};

342  Reference_base()

346  link(new Referenced_data(

360  Ref_any& operator= (Ref_any&& other) const noexcept
361  {
362  // self-copy: nothing to do
363  if (&other == this) return *this;
364  // we'll be copied into, start nullifying ourselves first
365  reset();
366  // if the other is null also, we're done
367  if (other.is_null()) return *this;
368  // the other ref notification disappears
369  other.m_content->m_buffer->m_notifications.erase(&other);
370  // since we get the same privileges as those we release we can just steal the content
371  m_content = other.m_content;
372  other.m_content = nullptr;
373  return *this;
374  }

473  Ref copy() const

475  return do_copy(*this);
476  }
477 
484  void* release() noexcept
485  {
486  if (is_null()) return nullptr;
487 
488  // notify everybody of the nullification
489  while (!m_content->m_buffer->m_notifications.empty()) {
490  // get the key of a notification
491  const Reference_base* key = m_content->m_buffer->m_notifications.begin()->first;
492  // call this notification, this might invalidate any iterator
493  m_content->m_buffer->m_notifications.begin()->second(*this);
494  // remove the notification we just called
495  m_content->m_buffer->m_notifications.erase(key);
496  }
497 
498  void* result = m_content->m_data;
499  m_content->m_data = nullptr;
500  m_content->m_buffer->m_delete = []() {}; // Referenced_metadata won't delete data
501 
502  unlink();
503 
504  return result;
505  }
506 
511  void on_nullify(std::function<void(Ref)> notifier) const noexcept
512  {
513  if (!is_null()) m_content->m_buffer->m_notifications[this] = notifier;
514  }
515 
516 private:
524  bool PDI_NO_EXPORT is_null() const noexcept
525  {
526  if (!m_content) return true;
527  if (!m_content->m_data) {
528  unlink();
529  return true;
530  }
531  return false;
532  }
533 
538  void PDI_NO_EXPORT unlink() const noexcept
539  {
540  assert(m_content);
541  m_content->m_buffer->m_notifications.erase(this);
542  if (R || W) --m_content->m_buffer->m_write_locks;
543  if (W) --m_content->m_buffer->m_read_locks;
544  --m_content->m_owners;
545  if (!m_content->m_owners) delete m_content;
546  m_content = nullptr;
547  }
548 
556  void PDI_NO_EXPORT link(Referenced_data* content) noexcept
557  {
558  assert(!m_content);
559  if (!content || !content->m_data) return; // null ref
560  if (R && content->m_buffer->m_read_locks) return;
561  if (W && content->m_buffer->m_write_locks) return;
562  m_content = content;
563  ++m_content->m_owners;
564  if (R || W) ++m_content->m_buffer->m_write_locks;
565  if (W) ++m_content->m_buffer->m_read_locks;
566  }
