185 Referenced_data(void* data, std::function<void(void*)> freefunc, Datatype_sptr type, bool readable, bool writable):
186 m_data{data},
187 m_type{type},
188 m_owners{0}
189 {
190 assert(data);
191 m_buffer = new Referenced_buffer{[data, freefunc, type]()

193 type->destroy_data(data);
194 freefunc(data);
195 }, readable, writable};
196 m_buffer->m_owners++;

206 {
207 m_buffer->m_owners--;
208 if (m_buffer->m_owners == 0) {

313 Ref_any(Ref_any&& other) noexcept:
314 Reference_base()
315 {
316 if (!other.m_content) return;
317 // the other ref notification disappears
318 other.m_content->m_buffer->m_notifications.erase(&other);
319 // since we get the same privileges as those we release we can just steal the content
320 m_content = other.m_content;
321 other.m_content = nullptr;
322 }

332 Ref_any(void* data, std::function<void(void*)> freefunc, Datatype_sptr type, bool readable, bool writable):
333 Reference_base()
334 {
335 if (type->datasize() && !data && (readable||writable)) {
336 throw Type_error{"Referencing null data with non-null size"};
337 }
338 if (data) {
339 link(new Referenced_data(data, freefunc, std::move(type), readable, writable));
340 }
341 }

350 Ref_any& operator= (Ref_any&& other) noexcept
351 {
352 // self-copy: nothing to do
353 if (&other == this) return *this;
354 // we'll be copied into, start nullifying ourselves first
355 reset();
356 // if the other is null also, we're done
357 if (other.is_null()) return *this;
358 // the other ref notification disappears
359 other.m_content->m_buffer->m_notifications.erase(&other);
360 // since we get the same privileges as those we release we can just steal the content
361 m_content = other.m_content;
362 other.m_content = nullptr;
363 return *this;
364 }

367 {
368 // self-copy: nothing to do
369 if (&other == this) return *this;
370 // we'll be copied into, start nullifying ourselves first
371 reset();
372 // and copy the content from the other
373 link(get_content(other));
374 return *this;
375 }

513 template<class T>
514 T scalar_value() const
515 {
516 static_assert(R, "Cannot get scalar_value from Ref without read access");
517 if (auto&& scalar_type = std::dynamic_pointer_cast<const Scalar_datatype>(type())) {
518 if (scalar_type->kind() == PDI::Scalar_kind::UNSIGNED) {
519 switch (scalar_type->buffersize()) {
520 case 1L:
521 return *static_cast<const uint8_t*>(m_content->m_data);
522 case 2L:
523 return *static_cast<const uint16_t*>(m_content->m_data);
524 case 4L:
525 return *static_cast<const uint32_t*>(m_content->m_data);
526 case 8L:
527 return *static_cast<const uint64_t*>(m_content->m_data);
528 default:
529 throw Type_error{"Unknown size of unsigned integer datatype"};
530 }
531 } else if (scalar_type->kind() == PDI::Scalar_kind::SIGNED) {
532 switch (scalar_type->buffersize()) {
533 case 1L:
534 return *static_cast<const int8_t*>(m_content->m_data);
535 case 2L:
536 return *static_cast<const int16_t*>(m_content->m_data);
537 case 4L:
538 return *static_cast<const int32_t*>(m_content->m_data);
539 case 8L:
540 return *static_cast<const int64_t*>(m_content->m_data);
541 default:
542 throw Type_error{"Unknown size of integer datatype"};
543 }
544 } else if (scalar_type->kind() == PDI::Scalar_kind::FLOAT) {
545 switch (type()->buffersize()) {
546 case 4L: {
547 return *static_cast<const float*>(m_content->m_data);
548 }
549 case 8L: {
550 return *static_cast<const double*>(m_content->m_data);
551 }
552 default:
553 throw Type_error{"Unknown size of float datatype"};
554 }
555 } else {
556 throw Type_error{"Unknown datatype to get value"};
557 }
558 }
559 throw Type_error{"Expected scalar, found invalid type instead: {}", type()->debug_string()};
560 }

573 void reset() noexcept

583 Ref copy() const
584 {
585 return do_copy(*this);
586 }
587
594 void* release() noexcept
595 {
596 if (is_null()) return nullptr;
597
598 // notify everybody of the nullification
599 while (!m_content->m_buffer->m_notifications.empty()) {
600 // get the key of a notification
601 const Reference_base* key = m_content->m_buffer->m_notifications.begin()->first;
602 // call this notification, this might invalidate any iterator
603 m_content->m_buffer->m_notifications.begin()->second(*this);
604 // remove the notification we just called
605 m_content->m_buffer->m_notifications.erase(key);
606 }
607
608 void* result = m_content->m_data;
609 m_content->m_data = nullptr;
610 m_content->m_buffer->m_delete = []() {}; // Referenced_metadata won't delete data
611
612 unlink();
613
614 return result;
615 }
616
621 void on_nullify(std::function<void(Ref)> notifier) const noexcept
622 {
623 if (!is_null()) m_content->m_buffer->m_notifications[this] = notifier;
624 }
625
626private:
634 bool PDI_NO_EXPORT is_null() const noexcept
635 {
636 if (!m_content) return true;
637 if (!m_content->m_data) {
638 unlink();
639 return true;
640 }
641 return false;
642 }
643
648 void PDI_NO_EXPORT unlink() const noexcept
649 {
650 assert(m_content);
651 m_content->m_buffer->m_notifications.erase(this);
652 if (R || W) --m_content->m_buffer->m_write_locks;
653 if (W) --m_content->m_buffer->m_read_locks;
654 --m_content->m_owners;
655 if (!m_content->m_owners) delete m_content;
656 m_content = nullptr;
657 }
658
667 void PDI_NO_EXPORT link(Referenced_data* content) noexcept
668 {
669 assert(!m_content);
670 if (!content || !content->m_data) return; // null ref
671 if (R && content->m_buffer->m_read_locks || W && content->m_buffer->m_write_locks) {
672 if (content->m_owners == 0) delete content;
673 return;
674 }
675 m_content = content;
676 ++m_content->m_owners;
677 if (R || W) ++m_content->m_buffer->m_write_locks;
678 if (W) ++m_content->m_buffer->m_read_locks;
679 }
680
681};
